Flask + Neo4j Project Structure

1. Directory Structure

/flask_neo4j_app
│── /app
│   │── /models           # Neo4j database models (Define node and relationship structures)
│   │── /routes           # Flask API routes (Define API endpoints for interacting with Neo4j)
│   │── /utils            # Utility functions (Database connections, helpers, logging, etc.)
│   │── __init__.py       # Initializes the app module
│── /config               # Configuration files (Environment variables, credentials, settings)
│── /tests                # Unit and integration tests (Test API and database functionality)
│── main.py               # Entry point (Initializes Flask and registers blueprints)
│── requirements.txt      # Dependencies (List of required Python packages)
│── .env                  # Environment variables (Neo4j URI, user credentials, debug mode)

2. Database Connection (`utils/db.py`)

from neo4j import GraphDatabase
import os

NEO4J_URI = os.getenv("NEO4J_URI", "bolt://localhost:7687")
NEO4J_USER = os.getenv("NEO4J_USER", "neo4j")
NEO4J_PASSWORD = os.getenv("NEO4J_PASSWORD", "password")

driver = GraphDatabase.driver(NEO4J_URI, auth=(NEO4J_USER, NEO4J_PASSWORD))

def query_neo4j(cypher_query, parameters=None):
    try:
        with driver.session() as session:
            return session.run(cypher_query, parameters)
    except Exception as e:
        print(f"Database query failed: {e}")
        return None

3. API Routes (`routes/person.py`)

from flask import Blueprint, request, jsonify
from app.utils.db import query_neo4j

person_bp = Blueprint("person", __name__)

@person_bp.route("/add_person", methods=["POST"])
def add_person():
    data = request.json
    name = data.get("name", "").strip()
    if not name or len(name) > 50:
        return jsonify({"error": "Invalid name"}), 400
    query = "CREATE (p:Person {name: $name}) RETURN p"
    result = query_neo4j(query, {"name": name})
    return jsonify({"message": "Person added"})

4. Application Entry Point (`main.py`)

from flask import Flask
from app.routes.person import person_bp
from app.utils.db import close_connection
import os

app = Flask(__name__)
app.register_blueprint(person_bp, url_prefix="/api")

@app.route("/")
def home():
    return "Flask + Neo4j API Running!"

@app.teardown_appcontext
def shutdown_app(exception=None):
    close_connection()

if __name__ == "__main__":
    DEBUG_MODE = os.getenv("FLASK_DEBUG", "False").lower() == "true"
    app.run(debug=DEBUG_MODE)
